{
	vector<pair<int, int>> res;
	for (auto s : arr)
	{
		int begintime = (s[0] - '0') * 600 + (s[1] - '0') * 60 + (s[3] - '0') * 10 + (s[4] - '0');

		{

{
	int healthy = heal;
	for (auto & ve : mob)
	{
		int mob_heal = ve[2];
		int damage = max(ve[0] - B,0);
		int attack = max(A - ve[1], 0);

			mob_heal -= attack;
			healthy -= damage;

	}
